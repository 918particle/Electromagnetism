\section{Memory Bank}
\begin{enumerate}
\item Displacement current: $\mathbf{J}_{\rm D} = \epsilon_0 \frac{\partial \mathbf{E}}{\partial t}$
\item Amp\`{e}re's Law: $\nabla \times \mathbf{B} = \mu_0 \mathbf{J} + \mu_0 \epsilon_0 \frac{\partial \mathbf{E}}{\partial t}$
\end{enumerate}
