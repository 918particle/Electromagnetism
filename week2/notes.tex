\section{Laplace's Equation}

\begin{frame}{Laplace's Equation in 1D}
\begin{equation}
\nabla^2 V = 0
\end{equation}
\begin{equation}
\frac{d^2V}{dx^2} = 0
\end{equation}
This implies:
\begin{equation}
V(x) = a x + V_0
\end{equation}
\end{frame}
